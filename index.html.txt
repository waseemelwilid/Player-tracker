  <script>
    const boxes = document.querySelectorAll(".habitBox");
    const dailyXpEl = document.getElementById("dailyXp");
    const totalXpEl = document.getElementById("totalXp");
    const levelEl = document.getElementById("level");
    const saveBtn = document.getElementById("saveBtn");

    let totalXp = Number(localStorage.getItem("totalXp")) || 0;
    let todaySaved = localStorage.getItem("todaySaved") === new Date().toDateString();

    function getLevel(xp) {
      if (xp < 200) return "Foundation";
      if (xp < 500) return "Consistency";
      if (xp < 1000) return "Discipline";
      if (xp < 2000) return "Advanced";
      if (xp < 3500) return "Elite Habits";
      return "1% Mindset";
    }

    function updateXP() {
      let dailyXp = 0;
      boxes.forEach(b => {
        if (b.checked) dailyXp += 10;
      });

      dailyXpEl.textContent = dailyXp;
      totalXpEl.textContent = totalXp + dailyXp;
      levelEl.textContent = getLevel(totalXp + dailyXp);

      saveBtn.disabled = dailyXp === 0 || todaySaved;
    }

    boxes.forEach(box => box.addEventListener("change", updateXP));

    saveBtn.addEventListener("click", () => {
      totalXp += Number(dailyXpEl.textContent);
      localStorage.setItem("totalXp", totalXp);
      localStorage.setItem("todaySaved", new Date().toDateString());
      todaySaved = true;
      saveBtn.disabled = true;
      alert("XP saved. Come back tomorrow.");
    });

    totalXpEl.textContent = totalXp;
    levelEl.textContent = getLevel(totalXp);
    if (todaySaved) saveBtn.disabled = true;
  </script>
